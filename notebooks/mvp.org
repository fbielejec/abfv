* Pipeline overview

- *Step 1:* generate the Fv 3D structure with ABodyBuilder3 -> ~out/complex.pdb~.
- *Step 2* isolate chains
- *Step 3:* run FreeSASA on the complex and each isolated chain => returns per-residue side-chain SASA.
- *STEP 4*: Compute the ΔSASA = isolated - complex / isolated and apply the >= 10% threshold (aka contact definition)
- *STEP 5*: visualize

def split_chain(src_pdb_path, chain, dst_pdb_path):
    with open(src_pdb_path) as fr:
        lines = fr.readlines()

    kept = [line for line in lines if line.startswith(("ATOM", "HETATM", "TER")) and len(line) > CHAIN_COLUMN and line[CHAIN_COLUMN] == chain]

    with open(dst_pdb_path, "w") as fw:
        fw.writelines(kept)
        fw.write("END\n")

    return dst_pdb_path

# STEP 3

def run_freesasa(pdb):
    proc = subprocess.run([FREESASA, "--format=rsa", pdb],
                         capture_output=True, text = True, check = True)
    return proc.stdout

def parse_rsa(rsa_text):
    """return {(chain, number, name) :  value}"""
    residues = {}
    for line in rsa_text.splitlines():
        if not line.startswith("RES"):
            continue
        cols = line.split()

        residue_name, chain, residue_number = cols[1],cols[2], int(cols[3])

        side_chain_absolute = float(cols[6])
        side_chain_relative = float(cols[7]) if cols[7] not in ("N/A", "-") else math.nan

        residues[(chain, residue_name, residue_number)] = {"side_chain_absolute": side_chain_absolute, "side_chain_relative": side_chain_relative}
    return residues

* STEP 4 — ΔSASA + contacts

#+BEGIN_SRC jupyter-python :session abfv :kernel abfv :async yes :exports both
# STEP 4

# merge dicts
isolated_rsa = {**heavy_rsa, **light_rsa}

contacts = []

# (iso_ABS - complex_ABS) / iso_ABS
# or
# iso_REL - complex_REL

for key in complex_rsa.keys():
    chain, residue_name, residue_number = key
    delta_abs = isolated_rsa[key]["side_chain_absolute"] - complex_rsa[key]["side_chain_absolute"]
    delta_rel = isolated_rsa[key]["side_chain_relative"] - complex_rsa[key]["side_chain_relative"]

    # delta <= 0 => no burial => not a contact. Mark both metrics NaN (not 0) to mirror the
    # Rust pipeline's convention. delta_rel > 0 iff delta_abs > 0 (same per-type reference),
    # so one flag gates both; it also guards the metric_abs division when iso_abs == 0.
    buried = delta_abs > 0
    metric_abs = delta_abs / isolated_rsa[key]["side_chain_absolute"] if buried else math.nan
    metric_rel = delta_rel / 100 if buried else math.nan

    contacts.append(dict(
        chain = chain,
        residue_number = residue_number,
        residue_name = residue_name,
        side_chain_absolute_iso = isolated_rsa[key]["side_chain_absolute"],
        side_chain_relative_iso = isolated_rsa[key]["side_chain_relative"],
        side_chain_absolute_complex = complex_rsa[key]["side_chain_absolute"],
        side_chain_relative_complex = complex_rsa[key]["side_chain_relative"],
        metric_abs = metric_abs,
        metric_rel = metric_rel
    ))

pprint.pp(contacts[0:2])

df = pd.DataFrame(contacts).sort_values(["chain", "residue_number"]).reset_index(drop = True)
df["is_contact_rel"] = df["metric_rel"] >= CONTACT_THRESHOLD
df["is_contact_abs"] = df["metric_abs"] >= CONTACT_THRESHOLD

* STEP 5 — visualize

#+BEGIN_SRC jupyter-python :session abfv :kernel abfv :async yes :exports both
# STEP 5

def visualize(df, metric):
    figure, axes = plt.subplots(2, 1, figsize=(13, 6), constrained_layout=True)

    for axis, chain in zip(axes, ["H", "L"]):
        sub_df = df[df.chain == chain]
        colors = ["#ff6347" if c else "#4787ff" for c in sub_df[metric] > CONTACT_THRESHOLD]

        axis.bar(sub_df["residue_number"], sub_df[metric].fillna(0),
           color=colors, width=0.9)

        axis.axhline(CONTACT_THRESHOLD, color="black", ls="--", lw=1)
        axis.set_title(f"{chain} chain")
        axis.set_ylabel("Δ side-chain SASA")
        axis.set_xlabel("residue #")
        axis.margins(x=0.01)

    figure.suptitle(f"{metric.upper()} (threshold {CONTACT_THRESHOLD*100:.0f}%)")

visualize(df, "metric_rel")
visualize(df, "metric_abs")
